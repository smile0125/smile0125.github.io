/\*\*

- 手动实现迭代器
- @param {list} 传入的集合
  \*/

```
function iteratorGenerator (list) {
   // 传入的集合的长度
   let length = list.length;

   // 记录当前访问的索引
   let idx = 0

   return {
       next: function () {
           // 如果索引超过集合的长度则为false
           const done = idx >= length;

           // 如果done为false，则可以继续取值
           const value = !done ? list[idx++] : undefined;

           // 返回当前的值
           return {
               done,
               value
           }
       }
   }
}
var iterator = iteratorGenerator(['1号选手', '2号选手', '3号选手'])
console.log(iterator.next())
console.log(iterator.next())
console.log(iterator.next())
console.log(iterator.next())
```
